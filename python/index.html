def calculate_suit_power(base_power, boost=1.0, *upgrades, **specs):
    """
    Calculate Iron Man suit's total power level.
    
    Args:
        base_power: Core arc reactor output (int)
        boost: Power multiplier (float, default 1.0)
        *upgrades: Variable number of upgrade names
        **specs: Keyword specifications (mode="stealth")
    Returns:
        float: Total calculated power
    """
    total = base_power * boost
    total += len(upgrades) * 10   # each upgrade adds 10
    print(f"Mode: {specs.get('mode', 'standard')}")
    return total

from abc import ABC, abstractmethod

# ─── ABSTRACT BASE CLASS ─────────────────────────────
class Hero(ABC):
    def __init__(self, name: str, power: int):
        self.name = name
        self._power = power          # _ means "protected"
        self.__secret_id = None    # __ means "private"

    @property
    def power(self):
        return self._power

    @power.setter
    def power(self, value):
        if value < 0: raise ValueError("Power can't be negative!")
        self._power = value

    @abstractmethod               # MUST be overridden by children
    def attack(self):
        pass

    def __str__(self):             # Magic method for print()
        return f"Hero: {self.name} | Power: {self._power}"

# ─── INHERITANCE ─────────────────────────────────────
class IronMan(Hero):
    suit_count = 0              # Class variable — shared

    def __init__(self, name, power, suit_model):
        super().__init__(name, power)   # call parent
        self.suit_model = suit_model
        IronMan.suit_count += 1

    def attack(self):            # Required override
        return f"⚡ {self.name} fires repulsor blast!"

    @classmethod
    def get_suit_count(cls):
        return f"{cls.suit_count} suits built"

tony = IronMan("Tony", 100, "Mark L")
print(tony)                    # Uses __str__
print(tony.attack())
tony.power = 150               # Uses @property setter

# ─── CLEAN API WRAPPER CLASS ──────────────────────────
class WeatherIntel:
    """Weather intelligence module for JARVIS"""
    BASE_URL = "https://api.openweathermap.org/data/2.5"

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.session = requests.Session()   # Reuse connection

    def get_weather(self, city: str) -> dict:
        try:
            resp = self.session.get(
                f"{self.BASE_URL}/weather",
                params={"q": city, "appid": self.api_key, "units": "metric"},
                timeout=10
            )
            resp.raise_for_status()   # Raises exception if 4xx/5xx
            data = resp.json()
            return {
                "city": data["name"],
                "temp": data["main"]["temp"],
                "feels_like": data["main"]["feels_like"],
                "condition": data["weather"][0]["description"],
                "humidity": data["main"]["humidity"]
            }
        except requests.exceptions.ConnectionError:
            raise ConnectionError("Network uplink failed")
        except requests.exceptions.Timeout:
            raise TimeoutError("Weather API timeout")

jarvis_weather = WeatherIntel(API_KEY)

        self._init_tables()

    def _init_tables(self):
        with self._get_conn() as conn:
            conn.execute("""
                CREATE TABLE IF NOT EXISTS expenses (
                    id       INTEGER PRIMARY KEY AUTOINCREMENT,
                    amount   REAL    NOT NULL,
                    category TEXT    NOT NULL,
                    note     TEXT,
                    date     TEXT    DEFAULT CURRENT_DATE
                )
            """)

    @contextmanager
    def _get_conn(self):
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row   # dict-like rows!
        try:
            yield conn
            conn.commit()
        finally:
            conn.close()

    def add_expense(self, amount, category, note=""):
        # ✅ Parameterized query — NEVER use f-strings in SQL!
        with self._get_conn() as conn:
            conn.execute(
                "INSERT INTO expenses (amount, category, note) VALUES (?, ?, ?)",
                (amount, category, note)
            )

    def get_by_category(self, category):
        with self._get_conn() as conn:
            rows = conn.execute(
                "SELECT * FROM expenses WHERE category = ? ORDER BY date DESC",
                (category,)
            ).fetchall()
        return [dict(r) for r in rows]

SECURITY ALERT: SQL Injection is the #1 database attack. NEVER do: f"SELECT * FROM users WHERE name = '{user_input}'" — a hacker types: ' OR 1=1; DROP TABLE users -- and your database is gone. ALWAYS use parameterized queries with ? placeholders.

import asyncio, aiohttp, functools, time
from typing import Callable, Any

# ─── DECORATOR FACTORY ───────────────────────────────
def retry(max_attempts: int = 3, delay: float = 1.0):
    """Retry decorator with configurable attempts"""
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)   # preserve function metadata
        def wrapper(*args, **kwargs) -> Any:
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1: raise
                    time.sleep(delay * (2 ** attempt))  # exponential backoff
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.5)
def unreliable_api_call():
    pass   # auto-retries 3x on failure!

# ─── GENERATORS — MEMORY EFFICIENT ───────────────────
def fibonacci_stream():
    """Infinite Fibonacci generator — uses ZERO extra memory"""
    a, b = 0, 1
    while True:
        yield a         # pauses here, returns a, resumes on next()
        a, b = b, a + b

fib = fibonacci_stream()
first_10 = [next(fib) for _ in range(10)]
