🌳
FHS — Filesystem Hierarchy
/bin, /etc, /var, /tmp, /proc, /dev, /home, /usr — what lives where and why.
🔒
Permissions: rwx
chmod 755, chmod u+x. Owner/Group/Others. Read=4, Write=2, Execute=1. Octal notation.
👤
Ownership: chown
chown user:group file. Files owned by users. Understanding uid/gid. Root owns everything.
🔗
Hard & Soft Links
ln (hard link) vs ln -s (symlink). Inodes. Hard links share data, symlinks point to paths.
💾
Disk & Mounts
df -h (disk space), du -sh (directory size), mount, umount, /etc/fstab — storage management.
📁
Special Files
/dev/null, /dev/zero, /dev/random, /proc/cpuinfo, /proc/meminfo — virtual filesystems.
🔐
SUID / SGID / Sticky
chmod 4755 (SUID), chmod 2755 (SGID), chmod +t /tmp (sticky). Advanced permission bits.
📊
Inodes & stat
stat file.txt — see inode, permissions, timestamps, block count. Every file is an inode.
PERMISSIONS DEEP DIVE
BASH — PERMISSIONS
# ─── READING PERMISSIONS ───────────────────────────────
ls -la
# -rwxr-xr-- 1 root fsoc 4096 Jan 1 exploit.sh
# ↑ type=file  ↑ owner=rwx  ↑ group=r-x  ↑ others=r--

# ─── chmod — CHANGE PERMISSIONS ────────────────────────
chmod 755 script.sh          # rwxr-xr-x (standard executable)
chmod 644 config.txt         # rw-r--r-- (standard file)
chmod 600 private.key        # rw------- (owner only!)
chmod +x script.sh           # Add execute for everyone
chmod u+x,g-w,o-rwx file    # Symbolic notation
chmod -R 755 /var/www/html   # Recursive chmod

# ─── chown — CHANGE OWNERSHIP ──────────────────────────
chown user:group file.txt
chown -R www-data:www-data /var/www/
chown root file              # Change only user
chown :devops file           # Change only group

# ─── SPECIAL PERMISSION BITS ───────────────────────────
chmod 4755 /usr/bin/passwd   # SUID — runs as owner
find / -perm -4000 2>/dev/null # Find ALL SUID files
chmod +t /tmp                # Sticky — only owner can delete

# ─── FILESYSTEM INSPECTION ─────────────────────────────
df -h                        # Disk usage all partitions
du -sh /var/log/*            # Size of each log file
stat /etc/passwd             # Full inode info
lsblk                        # Block devices (drives)
mount | column -t            # All mounted filesystems

# ─── SPECIAL FILES ─────────────────────────────────────
cat /proc/cpuinfo            # CPU info from kernel
cat /proc/meminfo            # Memory info from kernel
ls /dev/sd*                  # Storage devices
dd if=/dev/urandom bs=1M count=10 of=test.bin # Create 10MB random file

BASH — TEXT TOOLS
# ─── VIM ESSENTIAL COMMANDS ────────────────────────────
# i       — enter Insert mode
# Esc     — return to Normal mode
# :w      — save | :q — quit | :wq — save & quit | :q! — force quit
# dd      — delete line | yy — yank (copy) line | p — paste
# /word   — search | n — next match | N — prev match
# :%s/foo/bar/g — replace ALL foo with bar in file
# gg — go top | G — go bottom | :42 — go to line 42

# ─── GREP — FIND TEXT IN FILES ─────────────────────────

grep -i "ERROR" /var/log/syslog         # Case-insensitive

grep -v "DEBUG" app.log                 # Invert — exclude pattern
grep -E "^[0-9]{1,3}\.[0-9]{1,3}" access.log # Regex: IP addresses
grep -c "404" access.log                # Count matches

# ─── SED — EDIT WITHOUT OPENING ────────────────────────
sed 's/http/https/g' config.txt         # Replace in output
sed -i 's/old_pass/new_pass/g' .env     # Edit file in-place
sed -n '10,20p' bigfile.txt             # Print lines 10-20
sed '/^#/d' config.txt                  # Delete comment lines
sed 's/[[:space:]]*$//' file.txt        # Remove trailing spaces

# ─── AWK — THE DATA EXTRACTION ENGINE ──────────────────

awk -F: '{print $1, $3}' /etc/passwd   # username + uid

awk '{sum+=$5} END {print sum}' data   # Sum column 5
awk 'NR%2==0' file                      # Print even lines only

# ─── CUT / SORT / UNIQ COMBO ───────────────────────────

BASH SCRIPTING PATTERNS
BASH SCRIPT TEMPLATE
#!/bin/bash
# ════════════════════════════════════════════════
# Mr. Robot Deployment Script — Production Grade
# Usage: ./deploy.sh -e prod -v 2.1.0 -r
# ════════════════════════════════════════════════
set -euo pipefail             # Exit on error, undefined vars
IFS=$'\n\t'                   # Safer word splitting

# ─── COLORS ────────────────────────────────────
RED='\033[0;31m'   GREEN='\033[0;32m'
YELLOW='\033[1;33m' CYAN='\033[0;36m' NC='\033[0m'

# ─── LOGGING FUNCTIONS ─────────────────────────
log_info()  { echo -e "${GREEN}[INFO]${NC} $1"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC} $1"; }
log_error() { echo -e "${RED}[ERROR]${NC} $1" >&2; }

# ─── VARIABLES ────────────────────────────────
ENV=""  VERSION=""  ROLLBACK=false
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups/${TIMESTAMP}"

# ─── ARGUMENT PARSING ─────────────────────────
while getopts "e:v:rh" opt; do
  case $opt in
    e) ENV="$OPTARG" ;;
    v) VERSION="$OPTARG" ;;
    r) ROLLBACK=true ;;
    h) echo "Usage: $0 -e ENV -v VERSION [-r]"; exit 0 ;;
    ?) log_error "Unknown option"; exit 1 ;;
  esac
done

# ─── VALIDATION ───────────────────────────────
validate_env() {
  if [[ -z "$ENV" ]]; then
    log_error "Environment required. Use -e prod|staging"
    exit 1
  fi
}

# ─── CLEANUP ON EXIT ──────────────────────────
cleanup() { log_warn "Cleanup triggered"; }
trap cleanup EXIT

# ─── MAIN ─────────────────────────────────────
main() {
  validate_env
  log_info "Deploying v${VERSION} to ${ENV}"
  # Loop over servers in environment
  while IFS= read -r server; do
    log_info "Deploying to: $server"
    ssh "$server" "sudo systemctl restart app" || \
      log_warn "Failed on $server"
  done < "servers_${ENV}.txt"
}

main "$@"

# ─── SSH HARDENING — /etc/ssh/sshd_config ──────────────

PasswordAuthentication no      # Keys only — no passwords

# ─── CRONTAB SYNTAX ────────────────────────────────────
# ┌───── minute (0-59)
# │ ┌─── hour (0-23)
# │ │ ┌─ day of month (1-31)
# │ │ │ ┌ month (1-12)
# │ │ │ │ ┌ day of week (0=Sun)
# * * * * *  command

crontab -e                    # Edit your crontab
# Run backup every day at 2:30am
30 2 * * * /home/elliot/backup.sh >> /var/log/backup.log 2>&1
# Run every 5 minutes
*/5 * * * * /opt/monitor.sh
# Run on reboot
@reboot /opt/startup.sh
# Run first day of month at midnight
0 0 1 * * /opt/monthly-report.sh

# ─── SYSTEMD TIMER (better than cron!) ─────────────────
# Create /etc/systemd/system/backup.service
[Unit]
Description=Daily Backup Service

[Service]
Type=oneshot
ExecStart=/home/elliot/backup.sh
User=elliot

# Create /etc/systemd/system/backup.timer
[Unit]
Description=Run backup daily at 2:30am

[Timer]
OnCalendar=*-*-* 02:30:00
Persistent=true

[Install]
WantedBy=timers.target

sudo systemctl enable --now backup.timer
systemctl list-timers         # See all timers + next run
